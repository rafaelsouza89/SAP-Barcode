  METHOD  find_sub_bloq_for_related.
    " find related blocked instructions
    " there can be only one blocked instruction active per EXIDV

    DATA(ls_putaway) = me->get_data_putaway( ).

    CHECK ls_putaway-type = zcl_rf_constants=>gc_paway_type_block
   AND  ls_putaway-status = zcl_rf_constants=>gc_inst_stat_cancelled.


    DATA(lt_bl) = zcl_rf_inst_putaway=>query_blocked_instr(
        EXPORTING
          iv_lgnum     =  ls_putaway-lgnum
          iv_matnr     =  ls_putaway-matnr
      "    iv_exidv     =  ls_putaway-exidv
*          iv_transient = ABAP_TRUE
      ).

    LOOP AT lt_bl INTO DATA(ls_bl).
      LOOP AT gt_cont INTO DATA(ls_cont) WHERE id = ls_bl-id.
        INSERT ls_cont INTO TABLE rt_instr.
      ENDLOOP.
    ENDLOOP.



  ENDMETHOD.
