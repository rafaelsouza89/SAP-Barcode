  METHOD set_user_data.
    CHECK me->gs_data IS INITIAL.
    SELECT SINGLE a~lgnum, a~werks, a~lgort, b~uname, b~curr_printer,
      c~name AS curr_printer_name, tvswz~vstel
      FROM zrf_t320 AS a
      INNER JOIN zrf_lgnum_user AS b
      ON a~lgnum = b~lgnum
      LEFT JOIN tsp03d AS c
      ON c~padest = b~curr_printer
      LEFT JOIN tvswz AS tvswz
      ON tvswz~werks = a~werks
      INTO CORRESPONDING FIELDS OF @me->gs_data
      WHERE b~uname = @me->gv_uname.
    IF sy-subrc <> 0.
      RAISE EXCEPTION TYPE zcx_rf_exception
        EXPORTING
          textid = VALUE #( msgid = zcl_rf_constants=>gc_msgid msgno = '003' attr1 = me->get_uname( ) ).

    ENDIF.
  ENDMETHOD.
