  METHOD encode.

    DATA: lv_prefix TYPE zrf_prefix.
    DATA: lv_lgnum TYPE lgnum.

    CASE
    iv_obj_type .
      WHEN  zcl_rf_constants=>gc_bc_obj_type_mat.
        zcl_rf_matnr=>conv_int_to_ext(
          EXPORTING

        ).

        DATA(lv_num_obj) = CONV zrf_num_obj_type( zcl_rf_matnr=>build_ext_matnr(
          EXPORTING
            iv_partnr      = lv_part_nr    " Part Number
            iv_prefix      = lv_part_prefix    " Part Prefix
        ) ).
*      WHEN zcl_rf_constants=>gc_bc_obj_type_bin.
*        lv_num_obj = iv_num_obj_type.
*
*      WHEN zcl_rf_constants=>gc_bc_obj_type_serial.
*        lv_num_obj = iv_num_obj_type.
*
*      WHEN zcl_rf_constants=>gc_bc_obj_type_printer.
*        lv_num_obj = iv_num_obj_type.

      WHEN OTHERS.
        lv_num_obj = iv_num_obj_type.
    ENDCASE.

    SELECT SINGLE prefix
      FROM zrf_lgnum_prefx
      INTO lv_prefix
      WHERE lgnum = iv_lgnum
      AND obj_type = iv_obj_type.


    IF lv_prefix IS INITIAL.

      lv_lgnum = ' '.

      SELECT SINGLE prefix
      FROM zrf_lgnum_prefx
      INTO lv_prefix
      WHERE lgnum = lv_lgnum
      AND obj_type = iv_obj_type.


    ENDIF.


    CONCATENATE lv_prefix lv_num_obj INTO rv_barcode.
  ENDMETHOD.
